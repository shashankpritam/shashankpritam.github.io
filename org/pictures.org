<script>
(function() {
  var stack = document.getElementById('card-stack');
  if (!stack) return;
  var cards = Array.prototype.slice.call(stack.querySelectorAll('.card'));
  var n = cards.length;
  var current = 0;
  function render() {
    cards.forEach(function(card, i) {
      var rel = (i - current + n) % n;
      card.classList.remove('active', 'prev', 'next', 'hidden');
      if (rel === 0) card.classList.add('active');
      else if (rel === 1) card.classList.add('next');
      else if (rel === n - 1) card.classList.add('prev');
      else card.classList.add('hidden');
    });
  }
  function advance(dir) {
    current = (current + dir + n) % n;
    render();
  }
  stack.addEventListener('click', function(e) {
    var card = e.target.closest('.card');
    if (!card) return;
    if (card.classList.contains('next')) advance(1);
    else if (card.classList.contains('prev')) advance(-1);
  });
  document.addEventListener('keydown', function(e) {
    if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') return;
    if (e.key === 'ArrowRight') advance(1);
    if (e.key === 'ArrowLeft') advance(-1);
  });
  render();
})();
</script>
